CREATE TABLE guarulhos (
    NOME VARCHAR NOT NULL,
    LOCAL VARCHAR NOT NULL,
    ADMISSAO DATE,
    CARGO VARCHAR NOT NULL CHECK (CARGO <> ''),
    COMISSIONAMENTO VARCHAR,
    REMUNERACAO NUMERIC(12, 2) NOT NULL,
    ANTECIPACAO_SAL_FERIAS NUMERIC NOT NULL DEFAULT 0,
    HONORARIOS NUMERIC NOT NULL DEFAULT 0,
    ABONO NUMERIC NOT NULL DEFAULT 0,
    FERIAS_13O NUMERIC NOT NULL DEFAULT 0,
    CAT_COMPLEMENT NUMERIC NOT NULL DEFAULT 0,
    REDUTOR NUMERIC NOT NULL DEFAULT 0,
    DESCONTOS NUMERIC NOT NULL DEFAULT 0,
    LIQUIDO NUMERIC NOT NULL DEFAULT 0,
    REFERENCIA VARCHAR
);
